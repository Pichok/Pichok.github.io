def qui_quadrado(df, var1, var2, traducao=False):
    # Remover os registros ausentes de SEXO, TIPO_PARTO
    df2 = df[[var1,var2]].dropna(how = 'any', subset=[var1, var2])

    if traducao:
        df2[var1] = df2[var1].str.split(' / ').str[0]
    
    tabAbsoluto = pd.crosstab(df2[var1], df2[var2])
    tabRelativo = pd.crosstab(df2[var1], df2[var2], normalize=True)

    #print(df[var1].str.split(', ').explode().value_counts())
    
    tabela_combined = tabAbsoluto.astype(str) + ' (' + (tabRelativo * 100).round(2).astype(str) + '%)'

    
    tabela_combined.loc['Total'] = tabAbsoluto.sum().astype(str) + ' (' + (tabRelativo.sum() * 100).round(2).astype(str) + '%)'
    tabela_combined['Total'] = tabAbsoluto.sum(axis=1).astype(str) + ' (' + (tabRelativo.sum(axis=1) * 100).round(2).astype(str) + '%)'

    tabela_combined.at['Total', 'Total'] = tabAbsoluto.sum().sum().astype(str) + ' (' + (tabRelativo.sum().sum()* 100).round(2).astype(str) + '%)'

    # Calculando o qui-quadrado
    stat, p, dof, expected = stats.chi2_contingency(tabAbsoluto)
    
    print(f'Valor P: {p:.3f}')
    
    return tabela_combined
